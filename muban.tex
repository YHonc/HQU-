  \begin{algorithm}[H]
    \caption{示例算法：求数组最大值}
    \label{alg:example}
    \begin{algorithmic}[1]
    \REQUIRE 一个包含 $n$ 个元素的数组 $A$
    \ENSURE 数组 $A$ 的最大值
    \STATE 初始化 $max \gets A[1]$
    \FOR{$i \gets 2$ 到 $n$}
        \IF{$A[i] > max$}
            \STATE $max \gets A[i]$
        \ENDIF
    \ENDFOR
    \RETURN $max$
    \end{algorithmic}
    \end{algorithm}
